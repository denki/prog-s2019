** Teilaufgabe c
#+begin_src haskell
paths :: Tree a -> Tree [a]
paths = paths' []

paths' :: [a] -> Tree a -> Tree [a]
paths' prefix (Branch a s t)
  = Branch pr' (paths' pr' s) (paths' pr' t) where pr' = prefix ++ [a]
paths' prefix (Leaf a)
  = Leaf $ prefix ++ [a] 
#+end_src

** Teilaufgabe d
#+begin_src haskell
tmap :: (a -> b) -> Tree a -> Tree b
tmap f (Branch a s t) = Branch (f a) (tmap f s) (tmap f t)
tmap f (Leaf a)       = Leaf $ f a
#+end_src

 * Typeclassopedia

** Teilaufgabe b
#+begin_src haskell
map :: (a -> b) -> [a] -> [b]           -- 1
map _ []     = []                       -- 2
map f (x:xs) = f x : map f xs           -- 3

uncurry :: (a -> b -> c) -> (a, b) -> c -- 5
uncurry f (x, y) = f x y                -- 6

map (uncurry (+)) [(1,2), (3,4)]
=³ uncurry (+) (1,2) : map (uncurry (+)) [(3,4)]
=⁶ (+) 1 2 : map (uncurry (+)) [(3,4)]
=  3 : map (uncurry (+)) [(3,4)]
=³ 3 : uncurry (+) (3,4) : map (uncurry (+)) []
=⁶ 3 : (+) 3 4 : map (uncurry (+)) []
=  3 : 7 : map (uncurry (+)) []
=² 3 : 7 : []
≡  [3,7]
#+end_src
